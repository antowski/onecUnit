
Перем мСписокТестов;

Процедура Конструктор()
	мСписокТестов = СоздатьОбъект("СписокЗначений");
КонецПроцедуры // Конструктор

Процедура Добавить(пТест) Экспорт
	мСписокТестов.ДобавитьЗначение(пТест);
КонецПроцедуры // Добавить

Процедура Выполнить(пРезультат) Экспорт
	Для Инд = 1 По мСписокТестов.РазмерСписка() Цикл
		Тест = мСписокТестов.ПолучитьЗначение(Инд);
		Тест.Выполнить(пРезультат);
	КонецЦикла;	
КонецПроцедуры // Выполнить

Функция СоздатьТест(пИмяКлассаТестовогоСлучая, пИмяТеста) Экспорт
	
	Мета = СоздатьОбъект("MetaInfoClasses");
	Если Мета.КлассСуществует(пИмяКлассаТестовогоСлучая) = 0 Тогда
		ТекстИсключения = "Не обнаружен класс '" + пИмяКлассаТестовогоСлучая + "'";
		я()._ВыброситьИскл(ТекстИсключения);		
	КонецЕсли;
	
	Тест = СоздатьОбъект(пИмяКлассаТестовогоСлучая);
	Тест.Инит(пИмяТеста);
	
	Возврат Тест;
	
КонецФункции // СоздатьТест
