
Перем мКолЗапущено;
Перем мКолПровалено;
Перем мПодписчики;

Процедура Конструктор()
	мКолЗапущено = 0;
	мКолПровалено = 0;
	мПодписчики = СоздатьОбъект("СписокЗначений");
КонецПроцедуры // Конструктор

Процедура ЗарегистрироватьЗапуск(пТест) Экспорт
	
	мКолЗапущено = мКолЗапущено + 1;
	
	Для й = 1 По мПодписчики.РазмерСписка() Цикл
		Подписчик = мПодписчики.ПолучитьЗначение(й);
		Подписчик.ПриЗапуске(пТест);
	КонецЦикла;
	
КонецПроцедуры // ЗарегистрироватьЗапуск

Процедура ЗарегистрироватьПровал(пТест) Экспорт
	
	мКолПровалено = мКолПровалено + 1;
	
	Для й = 1 По мПодписчики.РазмерСписка() Цикл
		Подписчик = мПодписчики.ПолучитьЗначение(й);
		Подписчик.ПриПровале(пТест);
	КонецЦикла;
	
КонецПроцедуры // ЗарегистрироватьПровал

Процедура ЗарегистрироватьУспех(пТест) Экспорт
	
	Для й = 1 По мПодписчики.РазмерСписка() Цикл
		Подписчик = мПодписчики.ПолучитьЗначение(й);
		Подписчик.ПриУспешномЗавершении(пТест);
	КонецЦикла;
	
КонецПроцедуры // ЗарегистрироватьУспех

Процедура ДобавитьПодписчика(пОбъектПодписчик) Экспорт
	мПодписчики.ДобавитьЗначение(пОбъектПодписчик);
КонецПроцедуры // ДобавитьПодписчика
